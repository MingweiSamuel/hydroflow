107    #[expect(clippy::type_complexity, reason = "stream types with ordering")]
108    pub fn collect_quorum_with_response<
109        'a,
110        L: Location<'a> + NoTick,
111        Order,
112        K: Clone + Eq + Hash,
113        V: Clone,
114        E: Clone,
115    >(
116        responses: Stream<(K, Result<V, E>), Atomic<L>, Unbounded, Order>,
117        min: usize,
118        max: usize,
119    ) -> (
120        Stream<(K, V), Atomic<L>, Unbounded, Order>,
121        Stream<(K, E), Atomic<L>, Unbounded, Order>,
122    ) {
123        let tick = responses.atomic_source();
124        let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
125        let current_responses = not_all.chain(unsafe { responses.clone().tick_batch() });
126        let count_per_key = current_responses
127            .clone()
128            .fold_keyed_commutative(
129                q!(move || (0, 0)),
130                q!(
131                    move | accum, value | { if value.is_ok() { accum.0 += 1; } else {
132                    accum.1 += 1; } }
133                ),
134            );
135        let not_reached_min_count = count_per_key
136            .clone()
137            .filter_map(
138                q!(
139                    move | (key, (success, _error)) | if success < min { Some(key) } else
140                    { None }
141                ),
142            );
143        let reached_min_count = count_per_key
144            .clone()
145            .filter_map(
146                q!(
147                    move | (key, (success, _error)) | if success >= min { Some(key) }
148                    else { None }
149                ),
150            );
151        let just_reached_quorum = if max == min {
152            not_all_complete_cycle
153                .complete_next_tick(
154                    current_responses.clone().anti_join(reached_min_count),
155                );
156            current_responses.anti_join(not_reached_min_count)
157        } else {
158            let (min_but_not_max_complete_cycle, min_but_not_max) = tick.cycle();
159            let received_from_all = count_per_key
160                .filter_map(
161                    q!(
162                        move | (key, (success, error)) | if (success + error) >= max {
163                        Some(key) } else { None }
164                    ),
165                );
166            min_but_not_max_complete_cycle
167                .complete_next_tick(
168                    reached_min_count.filter_not_in(received_from_all.clone()),
169                );
170            not_all_complete_cycle
171                .complete_next_tick(
172                    current_responses.clone().anti_join(received_from_all),
173                );
174            current_responses.anti_join(not_reached_min_count).anti_join(min_but_not_max)
175        };
176        (
177            just_reached_quorum
178                .filter_map(
179                    q!(
180                        move | (key, res) | match res { Ok(v) => Some((key, v)), Err(_)
181                        => None, }
182                    ),
183                )
184                .all_ticks_atomic(),
185            responses
186                .filter_map(
187                    q!(
188                        move | (key, res) | match res { Ok(_) => None, Err(e) =>
189                        Some((key, e)), }
190                    ),
191                ),
192        )
193    }

268    type JoinResponses<K, M, V, L> = Stream<(K, (M, V)), Atomic<L>, Unbounded, NoOrder>;
269    /// Given an incoming stream of request-response responses, joins with metadata generated
270    /// at request time that is stored in-memory.
271    ///
272    /// The metadata must be generated in the same or a previous tick than the response,
273    /// typically at request time. Only one response element should be produced with a given
274    /// key, same for the metadata stream.
275    pub fn join_responses<
276        'a,
277        K: Clone + Eq + Hash,
278        M: Clone,
279        V: Clone,
280        L: Location<'a> + NoTick,
281    >(
282        tick: &Tick<L>,
283        responses: Stream<(K, V), Atomic<L>, Unbounded, NoOrder>,
284        metadata: Stream<(K, M), Tick<L>, Bounded, NoOrder>,
285    ) -> JoinResponses<K, M, V, L> {
286        let (remaining_to_join_complete_cycle, remaining_to_join) = tick
287            .cycle::<Stream<_, _, _, NoOrder>>();
288        let remaining_and_new: Stream<(K, M), Tick<L>, Bounded, _> = remaining_to_join
289            .chain(metadata);
290        let responses = unsafe { responses.tick_batch() };
291        let joined_this_tick = remaining_and_new
292            .clone()
293            .join(responses.clone())
294            .map(q!(| (key, (meta, resp)) | (key, (meta, resp))));
295        remaining_to_join_complete_cycle
296            .complete_next_tick(
297                remaining_and_new.anti_join(responses.map(q!(| (key, _) | key))),
298            );
299        joined_this_tick.all_ticks_atomic()
300    }
